Create `src/config/charsets.js`. This module owns all character-set toggle event listeners and the Last Active Guard UI.

```js
// src/config/charsets.js — Character set toggle controls with Last Active Guard
import { appState, triggerGenerate } from '../app.js';
import { isLastActive } from './guards.js';

// Maps checkbox id → appState enabledSets key
const SET_MAP = {
  'toggle-uppercase': 'uppercase',
  'toggle-lowercase': 'lowercase',
  'toggle-numbers':   'numbers',
  'toggle-symbols':   'symbols',
};

let guardErrorTimer = null;

/**
 * Shows the Last Active Guard inline error message, auto-dismisses after 3s.
 */
function showGuardError() {
  const el = document.getElementById('charset-error');
  if (!el) return;
  el.textContent = 'At least one character set must be active';
  el.style.display = 'block';
  if (guardErrorTimer) clearTimeout(guardErrorTimer);
  guardErrorTimer = setTimeout(() => {
    el.style.display = 'none';
  }, 3000);
}

/**
 * Hides the inline error immediately.
 */
function clearGuardError() {
  const el = document.getElementById('charset-error');
  if (el) el.style.display = 'none';
  if (guardErrorTimer) { clearTimeout(guardErrorTimer); guardErrorTimer = null; }
}

/**
 * Initialises the four character-set toggle checkboxes.
 * Must be called after DOMContentLoaded (elements must exist).
 */
export function initCharsetControls() {
  for (const [checkboxId, setKey] of Object.entries(SET_MAP)) {
    const checkbox = document.getElementById(checkboxId);
    if (!checkbox) continue;

    // Sync initial checked state from appState
    checkbox.checked = appState.enabledSets.includes(setKey);

    checkbox.addEventListener('change', () => {
      if (!checkbox.checked) {
        // User is trying to uncheck — guard check
        if (isLastActive(setKey, appState.enabledSets)) {
          // Block: restore the checkbox and show error
          checkbox.checked = true;
          showGuardError();
          return;
        }
        // Safe to remove
        clearGuardError();
        appState.enabledSets = appState.enabledSets.filter(k => k !== setKey);
      } else {
        // User is checking — always allowed
        clearGuardError();
        if (!appState.enabledSets.includes(setKey)) {
          appState.enabledSets = [...appState.enabledSets, setKey];
        }
      }
      triggerGenerate();
    });
  }
}
```

Key rules:
- Import `isLastActive` from `./guards.js` — do NOT inline the guard logic
- Import `appState` and `triggerGenerate` from `../app.js` — do NOT re-declare them
- Replace `appState.enabledSets` with a new array (filter/spread) rather than mutating in-place — this keeps the reference stable for `isLastActive` checks
- `showGuardError()` targets `#charset-error` element (added to HTML in Task 2)
- `guardErrorTimer` prevents error message from stacking on rapid clicks
- Do NOT call `triggerGenerate()` when blocking the last set (state did not change)
